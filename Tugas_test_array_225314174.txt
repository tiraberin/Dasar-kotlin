fun fibonacci(angka1: Int, angka2: Int, maks: Int): List<Int> {
    val fibonacciList = mutableListOf(angka1, angka2)

    while (true) {
        val nextNumber = fibonacciList[fibonacciList.size - 1] + fibonacciList[fibonacciList.size - 2]
        if (nextNumber > maks) {
            break
        }
        fibonacciList.add(nextNumber)
    }

    println("Deret Fibonacci: $fibonacciList")
    return fibonacciList
}

fun main(args: Array<String>){
    fibonacci(0, 1, 100)
}

fun text_to_array(text: String): Array<String> {
    val wordsArray = text.split(" ").toTypedArray()
    println("Array dari text: ${wordsArray.contentToString()}")
    return wordsArray
}

fun main(args: Array<String>){
    text_to_array("Hari senin saya belajar kotlin")
}
